CREATE TABLE DISTRIBUTOR
(
  DISTRIBUTOR_NAME VARCHAR(500) NOT NULL,
  PHONE_NUMBER NUMERIC(20) NOT NULL,
  STREET_ADDRESS VARCHAR(500) NOT NULL,
  CITY VARCHAR(500) NOT NULL,
  STATE VARCHAR(500) NOT NULL,
  ZIPCODE NUMERIC(10) NOT NULL,
  PRIMARY KEY (DISTRIBUTOR_NAME)
);

CREATE TABLE CRITICS
(
  CRITICS_NAME VARCHAR(500) NOT NULL,
  CRITIC_PHONE_NUMBER NUMERIC(20) NOT NULL,
  PRIMARY KEY (CRITIC_PHONE_NUMBER)
);

CREATE TABLE MOVIE
(
  MOVIE_TITLE VARCHAR(500) NOT NULL,
  DIRECTOR VARCHAR(500) NOT NULL,
  DISTRIBUTOR_NAME VARCHAR(500) NOT NULL,
  ACTORS VARCHAR(500) NOT NULL,
  RELEASE_DATE DATE NOT NULL,
  RATING INT NOT NULL,
  RUN_TIME INT NOT NULL,
  DISTRIBUTOR_NAME VARCHAR(500) NOT NULL,
  PRIMARY KEY (MOVIE_TITLE, DIRECTOR),
  FOREIGN KEY (DISTRIBUTOR_NAME) REFERENCES DISTRIBUTOR(DISTRIBUTOR_NAME)
);

CREATE TABLE DOMESTIC
(
  COMEDY_FLAG INT NOT NULL,
  DRAMA_FLAG INT NOT NULL,
  ACTION_FLAG INT NOT NULL,
  HORROR_FLAG INT NOT NULL,
  FUNNIENESS INT NOT NULL,
  DESCRIPTION VARCHAR(500) NOT NULL,
  MOVIE_TITLE VARCHAR(500) NOT NULL,
  DIRECTOR VARCHAR(500) NOT NULL,
  MOVIE_TITLE VARCHAR(500) NOT NULL,
  DIRECTOR VARCHAR(500) NOT NULL,
  FOREIGN KEY (MOVIE_TITLE, DIRECTOR) REFERENCES MOVIE(MOVIE_TITLE, DIRECTOR)
);

CREATE TABLE FOREIGN_TABLE
(
  MOVIE_TITLE VARCHAR(500) NOT NULL,
  LANGUAGE VARCHAR(500) NOT NULL,
  SUBTITLES VARCHAR(500) NOT NULL,
  DIRECTOR VARCHAR(500) NOT NULL,
  MOVIE_TITLE VARCHAR(500) NOT NULL,
  DIRECTOR VARCHAR(500) NOT NULL,
  FOREIGN KEY (MOVIE_TITLE, DIRECTOR) REFERENCES MOVIE(MOVIE_TITLE, DIRECTOR)
);

CREATE TABLE REVIEWED_BY
(
  MOVIE_TITLE VARCHAR(500) NOT NULL,
  DIRECTOR VARCHAR(500) NOT NULL,
  CRITIC_PHONE_NUMBER NUMERIC(20) NOT NULL,
  EVALUATION VARCHAR(500) NOT NULL,
  CRITIC_PHONE_NUMBER NUMERIC(20) NOT NULL,
  MOVIE_TITLE VARCHAR(500) NOT NULL,
  DIRECTOR VARCHAR(500) NOT NULL,
  FOREIGN KEY (CRITIC_PHONE_NUMBER) REFERENCES CRITICS(CRITIC_PHONE_NUMBER),
  FOREIGN KEY (MOVIE_TITLE, DIRECTOR) REFERENCES MOVIE(MOVIE_TITLE, DIRECTOR)
);
